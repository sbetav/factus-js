import { Callout } from "fumadocs-ui/components/callout";

## Firma del método

```ts
upload(input: UploadReceptionBillInput): Promise<ApiResponse<UploadedReceptionBill>>
```

## Endpoint Factus

`POST /v1/receptions/upload`

Fuente oficial: [Cargar factura](https://developers.factus.com.co/recepcion-de-documentos/cargar-factura/index.html)

## Parámetros

<TypeTable
  type={{
    
    track_id: {
      type: "string",
      required: true,
      description:
        "Track ID/CUFE de la factura emitida por el proveedor.",
    },

}}
/>

## Ejemplo

```ts
const response = await factus.reception.upload({
  track_id:
    "79760c1d956a143a1076c9d06808b0916f90eb3eec5d34697fd875a2f4be1c3c18ae583d902671161d443c5dfbd48a4d",
});

console.log(response.data.number);
```
