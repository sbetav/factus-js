import { Callout } from "fumadocs-ui/components/callout";

## Firma del método

```ts
sendEmail(
  number: string,
  input: SendCreditNoteEmailInput,
  options?: RequestOptions,
): Promise<SendCreditNoteEmailResponse>
```

## Endpoint Factus

`POST /v2/credit-notes/{number}/send-email`

Fuente oficial: [Enviar correo](https://developers.factus.com.co/notas-credito/enviar-correo)

## Ejemplo

```ts
await factus.creditNotes.sendEmail("NC856", {
  email: "alanturing@enigmasas.com",
});
```
