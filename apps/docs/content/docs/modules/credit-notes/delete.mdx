import { Callout } from "fumadocs-ui/components/callout";

## Firma del método

```ts
delete(referenceCode: string): Promise<DeleteCreditNoteResponse>
```

## Endpoint Factus

`DELETE /v1/credit-notes/reference/{reference_code}`

Fuente oficial: [Eliminar nota crédito](https://developers.factus.com.co/v1/notas-credito/eliminar/index.html)

<Callout type="warn">
  Solo se pueden eliminar notas crédito que no hayan sido validadas por la DIAN.
</Callout>

## Parámetros

<TypeTable
  type={{
    referenceCode: {
      type: "string",
      required: true,
      description: (
        <>
          {"Código de referencia ("}
          <code>{"reference_code"}</code>
          {") de la nota crédito no validada a eliminar."}
        </>
      ),
    },
  }}
/>
## Ejemplo

```ts
await factus.creditNotes.delete("NC856");
```
