import { Callout } from "fumadocs-ui/components/callout";

## Firma del método

```ts
delete(referenceCode: string): Promise<DeleteBillResponse>
```

## Endpoint Factus

`DELETE /v1/bills/destroy/reference/{reference_code}`

Fuente oficial: [Eliminar factura no validada](https://developers.factus.com.co/v1/facturas/eliminar/index.html)

<Callout type="warn">
  Solo se pueden eliminar facturas que no hayan sido validadas por la DIAN.
</Callout>

## Parámetros

<TypeTable
  type={{
    referenceCode: {
      type: "string",
      required: true,
      description: (
        <>
          {"Código de referencia ("}
          <code>{"reference_code"}</code>
          {") de la factura no validada a eliminar."}

      ),
    },
  }}
/>
## Ejemplo

```ts
await factus.bills.delete("SETP990000049");
```
