<Step>

## Configura las credenciales

Crea un archivo `.env.local` en la raíz de tu proyecto:

```bash title=".env.local"
FACTUS_CLIENT_ID=tu-client-id
FACTUS_CLIENT_SECRET=tu-client-secret
FACTUS_USERNAME=tu@email.com
FACTUS_PASSWORD=tu-contraseña
```

</Step>

<Step>

## Inicializa el cliente

```ts title="factus.ts"
import { FactusClient } from "factus-js";

export const factus = new FactusClient({
  clientId: process.env.FACTUS_CLIENT_ID!,
  clientSecret: process.env.FACTUS_CLIENT_SECRET!,
  username: process.env.FACTUS_USERNAME!,
  password: process.env.FACTUS_PASSWORD!,
  environment: "sandbox", // "production" para el entorno real
});
```

<Callout type="info">
  El SDK obtiene y renueva el token de acceso de forma automática por lo que no
  necesitas llamar a ningún método de inicio de sesión manualmente. Consulta la
  [documentación de autenticación](/docs/authentication) para más detalles.
</Callout>

</Step>

<Step>

## Crea tu primera factura

```ts
import factus from "./factus";
import {
  InvoiceDocumentType,
  OperationTypeCode,
  PaymentFormCode,
  PaymentMethodCode,
  IdentityDocumentTypeId,
  CustomerTributeId,
  OrganizationTypeId,
  ProductStandardId,
  FactusError,
} from "factus-js";

const createInvoice = async () => {
  try {
    const response = await factus.bills.create({
      numbering_range_id: 123,
      reference_code: "ORD-001",
      observation: "Factura de prueba",
      payment_form: PaymentFormCode.CashPayment,
      payment_method_code: PaymentMethodCode.Cash,
      payment_due_date: "2025-12-31",
      billing_period: {
        start_date: "2025-01-01",
        end_date: "2025-01-31",
      },
      customer: {
        identification: "900123456",
        dv: "7",
        company: "Empresa de Prueba S.A.S.",
        trade_name: "Empresa Prueba",
        names: "Juan",
        address: "Calle 123 # 45-67",
        email: "cliente@empresa.com",
        phone: "3001234567",
        legal_organization_id: OrganizationTypeId.LegalEntity,
        tribute_id: CustomerTributeId.IVA,
        identification_document_id: IdentityDocumentTypeId.NIT,
        municipality_id: 149, // Bogotá D.C.
      },
      items: [
        {
          code_reference: "PROD-001",
          name: "Producto de ejemplo",
          quantity: 1,
          discount_rate: 0,
          price: 100000,
          tax_rate: "19.00",
          unit_measure_id: 70, // unidad
          standard_code_id: ProductStandardId.TaxpayerAdoption,
          is_excluded: 0,
          tribute_id: 1,
        },
      ],
    });

    return response.data;
  } catch (error) {
    if (error instanceof FactusError) {
      console.error(error);
    }
  }
};
```
